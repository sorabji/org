** templates
example:

template <class T>
T GetMax (T a, T b)
{
  T result;
  result = (a>b) ? a : b;
  return (result);
}

int main()
{
  int i=4,j=3,k;
  long l=10,m=4,n;
  k=GetMax<int>(i,k);
  n=GetMax<long>(l,m);
  return 0;
}
   
at compile time, all instances of 'T' are replaced w/ the proper datatypes
